from machine import SoftI2C
i2c = SoftI2C(scl=Pin(22), sda=Pin(21))
oled = OLEDESP32.SSD1306_I2C(128, 32, i2c)
last_text = '' 
def OledPrint(text):
    global last_text
    if(text != last_text):
        last_text = text
        oled.fill(0)
        oled.show()
        words = text.split()
        lines = []
        current_line = ''
        remaining_space = 14
        for word in words:
            if len(word) + 1 <= remaining_space:
                current_line += word + ' '
                remaining_space -= len(word) + 1
            else:
                lines.append(current_line)
                current_line = word + ' '
                remaining_space = 14 - len(word) - 1
        lines.append(current_line)
        for i, line in enumerate(lines):
            oled.text(line, 0, i * 10)
            oled.show()
